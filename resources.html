Doing the data cleaning for the datasets

farmer_markets_clean = 
  farmer_markets %>% 
  janitor::clean_names() %>% 
  rename("Borough"= borough) %>% 
  drop_na()
gardens_clean = 
  gardens %>% 
  janitor::clean_names() %>% 
  rename("Borough" = boro) %>% 
  mutate(Borough = recode(Borough, "M"= "Manhattan",
                                "B"= "Brooklyn",
                                "X"= "Bronx",
                                "Q"= "Queens",
                                "S"= "Staten Island")) %>% 
  drop_na ()
healthy_stores_clean = 
  healthy_stores %>% 
  janitor::clean_names() %>% 
  rename("Borough"= borough) %>% 
  drop_na()
